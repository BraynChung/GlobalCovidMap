import time

lat_lon = []
geolocator = Nominatim(user_agent='app', timeout=10)

for location in df['Country']:
    time.sleep(1)
    location = geolocator.geocode(location)
    if location is None:
        lat_lon.append(np.nan)
    else:
        geo_coord = (location.latitude, location.longitude)
        lat_lon.append(geo_coord)
